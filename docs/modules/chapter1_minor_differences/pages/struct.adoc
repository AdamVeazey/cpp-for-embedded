typedef struct point {
    int x;
    int y;
} point_t;

point_t point_add(const point_t* const lhs, const point_t* const rhs) {
    point_t result;
    result.x = lhs->x + rhs->x;
    result.y = lhs->y + rhs->y;
    return result;
}

void foo(void) {
    point_t a = { .x = 5, .y = 10 };
    point_t b = { .x = 2, .y = 3 };
    point_t c = point_add( &a, &b );
}

        return Point( x + rhs.x, y + rhs.y);
